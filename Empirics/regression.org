def to_series(df):
    """
    Stack a dataframe to make a series.
    """
    try:
        df.columns.names = [n if n is not None else "_%d" % i for i,n in enumerate(df.columns.names)]
    except AttributeError: # Already a series?
        return pd.Series(df)

    for n in df.columns.names:
        df = df.stack(n)

    return df

def inner(a,b,idxout,colsout,fill_value=None,method='sum'):
    """
    Compute inner product of sorts, summing over indices of products which don't appear in idxout or colsout.
    """

    a = to_series(a)
    b = to_series(b)

    if fill_value is not None:
        a = a.astype(pd.SparseDtype(fill_value=fill_value))
        b = b.astype(pd.SparseDtype(fill_value=fill_value))

    idxint = list(set(a.index.names).intersection(b.index.names))
    aonly = list(set(a.index.names).difference(idxint))
    bonly = list(set(b.index.names).difference(idxint))

    if fill_value is None: # Non-sparse
        a = a.replace(0.,np.nan).dropna()
        b = b.replace(0.,np.nan).dropna()

    c = pd.merge(a.reset_index(aonly),b.reset_index(bonly),on=idxint)
    c = c.reset_index().set_index(idxint + aonly + bonly)

    sumover = list(set(aonly+bonly+idxint).difference(idxout+colsout))
    keep = list(set(aonly+bonly+idxint).difference(sumover))

    if fill_value is not None:
        foo = c.sparse.to_coo().tocsr()

        foo = foo[:,0].multiply(foo[:,1])
        foo = pd.DataFrame.sparse.from_spmatrix(foo,index=c.index)
    else:
        foo = c.iloc[:,0]*c.iloc[:,1]

    if method=='sum':
        p = foo.groupby(keep).sum()
    elif method=='mean':
        p = foo.groupby(keep).mean()
    else:
        raise ValueError("No method %s." % method)

    p = p.unstack(colsout)

    if len(idxout)>1:
        p = p.reorder_levels(idxout)

    p = p.sort_index()

    if len(colsout):
        p = p.sort_index(axis=1)

    return p

Provide a function =Mp= such that =Mp(X)= returns $X-\E(X|p)$.  In our
case we rely on the law of one price, so that this is a matter of
computing deviations of $X$ from time-market means of $X$.

import pandas as pd
import numpy as np
from warnings import warn

def Mp(X):
    """
    Construct X-E(X|p) = (I-S(S'S)^{-1}S')X.

    Drop any categorical variables where taking means isn't sensible.
    """
    if len(X.shape) > 1:
        X = X.loc[:,X.dtypes != 'category']
    else:
        if X.dtype == 'category': warn('Taking mean of categorical variable.')

    use = list(set(['t','m','j']).intersection(X.index.names))

    if len(use):
        return X - X.groupby(use).transform("mean")
    else:
        return X - X.mean()

** \(\mathcal{M}_d\) operator
Construct =Md= operator.  This can involve strictly categorical variables, in which case we can construct a non-parametric estimator of $\E(X|d=\tau)$ simply by taking means for every value of $\tau$ observed in the sample.    When the method is linear, we instead employ a linear projection.
#+begin_src python
def Md_generator(X,d,method='categorical',expected=False,Mpd=False):
    """
    Md operator, for either categorical or linear expectations.

    If expected is True, return E(X|d) rather than X-E(X|d).
    """
    if method=='categorical':
        try:
            groups = d.columns.tolist()
        except AttributeError: # d is a series?
            if d.name is None: d.name = 'tau'
            groups = [d.name]
        if 'j' in X.index.names: groups += 'j'
        if Mpd: groups += ['t','m']
        groups = list(set(groups)) # Eliminate dupes

        Xg = pd.DataFrame({'X':X}).join(d,how='left').groupby(groups)

        gamma_d = Xg.transform("mean").squeeze()
        gamma_d.name = 'gamma_d'

        if expected: return gamma_d,None

        MdX = X - gamma_d

    elif method=='linear':
        d = pd.DataFrame(d)

        # Add constant column if one doesn't exist in d
        d = d - d.mean()
        d['Constant'] = 1  # Constant just picks up means of X

        if 'j' in X.index.names:
            Xj = X.groupby('j',observed=True)
            if 'j' in d.index.names: dj = d.groupby('j')
            if expected: # Really return gamma_d,gamma
                EdX = {}
                Gamma = {}
                for j,y in Xj:
                    if 'j' in d.index.names:
                        x = dj.get_group(j).droplevel('j')
                    else:
                        x = d
                    EdX[j],Gamma[j] = Md_generator(y.droplevel('j'),x,method='linear',expected=True,Mpd=Mpd)
                EdX = pd.DataFrame(EdX)
                EdX.columns.name = 'j'
                Gamma = pd.DataFrame(Gamma).T
                Gamma.index.name = 'j'

                try:
                    EdX = EdX.stack()
                except AttributeError:
                    pass

                EdX = EdX.reorder_levels(['i','t','m','j']).sort_index()

                return EdX,Gamma
            else:
                MdX = {}
                for j,y in Xj:
                    if 'j' in d.index.names:
                        x = dj.get_group(j).droplevel('j')
                    else:
                        x = d
                    MdX[j] = Md_generator(y.droplevel('j'),x,
                                          method='linear',expected=False,Mpd=Mpd)
                MdX = pd.DataFrame(MdX)
                MdX.columns.name = 'j'

                try:
                    MdX = MdX.stack()
                except AttributeError:
                    pass

                MdX = MdX.reorder_levels(['i','t','m','j']).sort_index()

                return MdX
        else:
            y,x = drop_missing([X,d])

            if Mpd:
                y = y - y.groupby(['t','m']).transform('mean')
                x = x - x.groupby(['t','m']).transform('mean')

            gamma = np.linalg.lstsq(x,y,rcond=None)[0]

            gamma = pd.Series(gamma.squeeze(),index=x.columns)

            if expected:
                # NB: d, not x? Yes, because we can predict even missing values
                # this way.
                gamma_d = (d@gamma).squeeze()
                gamma_d.name = 'gamma_d'

                return gamma_d,gamma
            else:
                gamma_d = (x@gamma).squeeze()
                gamma_d.name = 'gamma_d'

            MdX = pd.Series(y.squeeze() - gamma_d,index=y.index)

    elif method=='mixed': # Some columns of d are categorical, others not
        if expected: raise NotImplementedError
        # Identify categorical columns
        cats = d.select_dtypes(['category']).columns
        xcols = d.columns.difference(cats)

        Z = X.join(d,how='outer')
        ycols = Z.columns.difference(xcols.union(cats))

        # Difference out means for each category
        if len(cats):
            Z = Z - Z.groupby(cats).transform("mean")
        else:
            Z = Z - Z.mean()

        y = Z[ycols]
        x = Z[xcols]

        MdX = Md_generator(y,x,method='linear') # Regress demeaned vars.

    else: raise ValueError("No method %s." % method)

    return MdX

def Mpd_generator(X,tau,method='categorical',Mpd=False):
    """
    Md operator, for either categorical or linear expectations.

    If Mpd is True, operator is Mpd (i.e. condition on d & p jointly).
    """

    if method=='categorical': # assuming conditioning is on groups tau
        groups = ['tau']
        if 'j' in X.index.names: groups += 'j'
        if Mpd: groups += ['t','m']
        Xg = pd.DataFrame({'X':X}).join(tau,how='left').groupby(groups)

        MdX = X - Xg.transform("mean").squeeze()
    elif method=='linear':
        try:
            taucols = tau.columns
            X = pd.DataFrame(X).join(tau,how='outer')
        except AttributeError:  # tau a Series
            taucols = tau

        if 'j' in X.index.names:
            MdX = X.groupby('j').apply(lambda y,x=taucols: Md_generator(y.droplevel('j'),x,method='linear',Mpd=Mpd)).T
            try:
                MdX = MdX.stack()
            except AttributeError:
                pass

            MdX = MdX.reorder_levels(['i','t','m','j']).sort_index()
        else:
            # Difference out kmeans if tau provided
            ycols = X.columns.difference(taucols)
            xcols = taucols
            group = []
            if 'tau' in tau: # kmeans categories provided
                xcols = taucols.drop('tau')
                group += 'tau'
            if Mpd: group += ['t','m']

            if len(group):
                X = X - X.groupby(group).transform("mean")
            else:
                X = X - X.mean()

            y = X[ycols]
            x = X[xcols]
            y,x = drop_missing([y,x])
            x['Constant'] = 1
            b = np.linalg.lstsq(x,y,rcond=None)[0]
            MdX = pd.Series(y.squeeze() - (x@b).squeeze(),index=y.index)
    else: raise ValueError("No method %s." % method)

    return MdX

def estimate_MpMdy(y,d,K=None,Mpd=False):

    if K is not None:
        d  = kmean_controls(K,Mp(y),Mp(d),classifiers=d.loc[:,d.dtypes == 'category'])
        method = 'categorical'
    else:
        method = 'linear'

        # Change categorical vars to numeric
        cats = d.select_dtypes(['category']).columns
        if len(cats):
            d[cats] = d[cats].apply(lambda x: x.cat.codes)

    Md = lambda x: Md_generator(x,d,method=method,Mpd=False)
    if Mpd:
        MpMd = lambda x: Md_generator(x,d,method=method,Mpd=True)
    else:
        MpMd = lambda x: Mp(Md(x))

    MpMdy = MpMd(y)

    assert MpMdy.index.names == ['i','t','m','j']

    if not np.all(np.abs(MpMdy.groupby(['j','t','m']).mean()) < 1e-6):
        warn('MpMdy means greater than 1e-6')

    return MpMdy,Md,MpMd,d

** Estimate \beta from $\MdMpy$

Given $\MpMdy$ estimate \beta.  The default approach to estimation here assumes a form of homoskedasticity.   But this is something we can test (and if necessary, correct).  The basic model involves a mean zero vector $X=\beta w + \epsilon$, from which we construct $\E XX^\T = \beta\E w^2\beta^\T + \E\epsilon\epsilon^\T$.  Under an assumption of homoskedasticity we have $\E\epsilon\epsilon^\T = \sigma^2\mI$.

def estimate_beta(MpMdy,
                  heteroskedastic=False,
                  cov = lambda X : pd.DataFrame.cov(X,ddof=0),
                  return_se=False,bootstrap_tol=None,verbose=False):

    With return_se=False (the default) this is a single cheap SVD.  With
    return_se=True and a bootstrap_tol it runs a resampling bootstrap of at
    least 30 iterations to get standard errors.

    WARNING (nested-bootstrap footgun): do not call with return_se=True
    inside an *outer* whole-pipeline bootstrap -- the inner >=30-iteration
    beta bootstrap then nests, blowing up cost ~30x with no benefit (the
    point estimate is all the outer loop needs).  In bootstrap loops pass
    return_se=False, or call Regression.get_beta with compute_se=False.
    """
    if verbose:
        print("estimate_beta")

    try:
        MpMdY = MpMdy.unstack('j')
    except KeyError:
        MpMdY = MpMdy

    C = cov(MpMdY)

    if np.any(np.isnan(C)):
        raise ValueError(f"Can't compute covariance matrix; too few {C.count().idxmin()}.")

    u,s,vt = svd_missing(C,max_rank=1,heteroskedastic=heteroskedastic)

    if np.sign(u).mean()<0: # Fix sign of u.
        u = -u

    b = pd.DataFrame(u*s,index=MpMdY.columns,columns=['beta'])

    if return_se and bootstrap_tol:
        if bootstrap_tol is None:
            raise ValueError("Not implemented. Specify bootstrap_tol>0.")
            V = (((e-e.mean())**2).mul(v**2,axis=0)).mean() # See p. 150 of Bai (2003)

            B = None

For each cell $c=(i,t,m)$ the welfare weight is the within-cell
projection of (residualized) log-expenditures onto \beta, computed over
the goods $O_c$ actually observed in that cell:
\[
   \hat w_c = \frac{\beta'X_c}{\beta'\beta}
            = \frac{\sum_{j\in O_c}\beta_j X_{cj}}{\sum_{j\in O_c}\beta_j^2}.
\]
Treating \beta as known (valid for inference on $w$ when $\sqrt{J}/T\to0$,
which holds here since the number of cells far exceeds the number of
goods; see [cite:@bai03]) the sampling variance has the sandwich form
\[
   \widehat{\Var}(\hat w_c)
      = \frac{\sum_{j\in O_c}\beta_j^2 \hat\sigma_{cj}^2}
             {\bigl(\sum_{j\in O_c}\beta_j^2\bigr)^2},
\]
where every sum ranges over the /same/ observed set $O_c$, so the result
is on the same scale as the estimated $\hat w_c$ (i.e. whatever
normalization \beta carries) and is consistent under missing data.  The
earlier expression divided a per-cell good-average by the global
$\beta'\beta$, which silently rescaled the variance by
$\overline{\beta_j^2}=\beta'\beta/J$ and mixed the observed-good count
with the full good count under missing data.

A subtlety the scale fix alone does not address: the residual $e_c$ is
not the raw error but the projection residual $e_c=M_\beta\epsilon_c$, with
$M_\beta=I-\beta\beta'/\beta'\beta$.  Because the $\beta$-direction has
already been fitted out, $\E[e_{cj}^2]=\sigma^2(1-h_{cj})$ with leverage
$h_{cj}=\beta_j^2/\sum_{k\in O_c}\beta_k^2$, so the raw HC0 estimate is
biased /downward/ by a finite-$|O_c|$ factor (about 20--25% at $J=8$;
$\to0$ as $J\to\infty$).  The correction is standard: divide by $(1-h_{cj})$
(HC2), which is exactly unbiased under homoskedasticity and remains
heteroskedasticity-robust.

How the per-good error variance $\sigma_{cj}^2$ is estimated is governed by
the =cov= argument:
- =hc0= (legacy default) :: White single residual, $\hat\sigma_{cj}^2=e_{cj}^2$.  No leverage correction; biased low at finite $J$.  Selected when =cov= is left unset, but with a =DeprecationWarning= --- it is retained as the default only for continuity with existing downstream users, and is slated to change to =hc2=.
- =hc1= :: HC0 scaled by $|O_c|/(|O_c|-1)$.
- =hc2= :: $e_{cj}^2/(1-h_{cj})$ --- exactly unbiased under homoskedasticity, still robust.  Recommended.
- =hc3= :: $e_{cj}^2/(1-h_{cj})^2$; jackknife-style, conservative.
- =homoskedastic= :: a single pooled $\hat\sigma^2=\sum e_{cj}^2/\sum_c(|O_c|-1)$ (proper degrees of freedom), reducing the formula to $\hat\sigma^2/\sum_{j\in O_c}\beta_j^2$.
- =pooled= :: $\hat\sigma_{cj}^2=\Var_i(e)$ within $(t,m,j)$ --- robust to heteroskedasticity across goods and markets, pooled across households so the per-cell estimate is stable (good for inverse-variance weighting downstream).  Not leverage-corrected, so it shares HC0's mild finite-$J$ downward bias.

All options share the same scale and missing-data handling; they differ
only in how $\hat\sigma_{cj}^2$ is estimated and corrected.

import warnings

def w_var(e,beta,cov=None):
    """Sampling variance of the estimated welfare weights w_hat_c.

    For each cell c=(i,t,m), w_hat_c = (beta'X_c)/(beta'beta) is the
    within-cell projection of residualized log-expenditures onto beta over
    the goods O_c observed in that cell.  Treating beta as known, the
    sandwich variance is

        Var(w_hat_c) = sum_{j in O_c} beta_j^2 sigma_cj^2
                       / (sum_{j in O_c} beta_j^2)^2,

    every sum over the same observed set so the result is on the same scale
    as w_hat_c and is consistent under missing data.

    Because e is the projection residual (the beta-direction is already
    fitted out), E[e_cj^2] = sigma^2 (1 - h_cj) with leverage
    h_cj = beta_j^2 / sum_{k in O_c} beta_k^2; the HC2/HC3 and homoskedastic
    options correct for this finite-J downward bias, HC0/HC1 do not.

    Parameters
    ----------
    e : Series or DataFrame
        Regression residuals (e.g. ``self.e3``); a Series indexed
        ``(i,t,m,j)`` or a wide DataFrame ``(i,t,m) x j``.
    beta : Series
        Frisch elasticities, indexed by good ``j``.
    cov : {'hc0','hc1','hc2','hc3','homoskedastic','pooled'}, optional
        Estimator for the per-good error variance sigma_cj^2.  Leaving it
        unset selects 'hc0' and raises a DeprecationWarning, because that
        default is biased low and is slated to change to 'hc2'; pass cov
        explicitly to silence the warning.

        - 'hc0' (legacy default) : White single residual, sigma_cj^2 = e_cj^2
                            (no leverage correction; biased low at finite J).
        - 'hc1'           : HC0 times |O_c|/(|O_c|-1).
        - 'hc2'           : e_cj^2/(1-h_cj); unbiased under homoskedasticity,
                            still robust.  Recommended.
        - 'hc3'           : e_cj^2/(1-h_cj)^2; conservative.
        - 'homoskedastic' : single pooled sigma^2 with proper d.o.f.
        - 'pooled'        : Var_i(e) within (t,m,j), pooled across
                            households (stable; shares HC0's finite-J bias).

        The 'homoskedastic' and 'pooled' options require residuals from
        which (t,m,j) groups can be formed (a Series, or a wide frame with
        t,m on the row index).

    Returns
    -------
    Series
        Estimated variance of ``w_hat_c`` for each cell ``c=(i,t,m)``, on
        the same scale as the estimated ``w``.
    """
    if cov is None:
        warnings.warn(
            "w_var default cov='hc0' is biased low at finite J (coverage ~0.78 "
            "with complete data, worse under missing goods) and is slated to "
            "change to 'hc2'.  Pass cov explicitly to silence this warning: "
            "'hc2' (recommended) or 'hc0' (legacy behavior).",
            DeprecationWarning, stacklevel=2)
        cov = 'hc0'

    if cov not in ('hc0','hc1','hc2','hc3','homoskedastic','pooled'):
        raise ValueError(f"Unknown cov={cov!r}; expected one of "
                         "'hc0','hc1','hc2','hc3','homoskedastic','pooled'.")

    E = e.unstack('j') if e.ndim==1 else e

    beta = pd.Series(beta).reindex(E.columns)
    B2 = (E*0).add(beta**2,axis=1)   # beta_j^2 where good observed, NaN where missing
    den = B2.sum(axis=1)             # Sum_{j in O_c} beta_j^2
    Jc = B2.notna().sum(axis=1)      # |O_c|, goods observed in cell
    h = B2.div(den,axis=0)           # leverage h_cj = beta_j^2 / sum_{O_c} beta_k^2

    # Var(w_hat_c) is unestimable when fewer than 2 goods are observed
    # (saturated fit, no residual d.o.f.); such cells return NaN.
    if cov=='homoskedastic':
        ssr = (E**2).sum().sum()
        sigma2 = ssr/(Jc-1).clip(lower=0).sum()          # pooled, proper d.o.f.
        return (sigma2/den).where(Jc>=2)

    if cov=='pooled':
        # sigma^2_{tmj} pooled across households -> stable per-cell estimate.
        # No leverage correction (it is unstable for small |O_c|), so this
        # shares HC0's mild finite-J downward bias; use 'hc2' if unbiasedness
        # matters more than estimator stability.
        S2 = E.groupby(['t','m']).transform('var')
        valid = E.notna() & np.isfinite(S2)
        num = (B2*S2).where(valid).sum(axis=1)
        d = B2.where(valid).sum(axis=1)
        return (num/d**2).where(Jc>=2)

    WE2 = E.multiply(beta,axis=1)**2                     # beta_j^2 e_cj^2
    if cov=='hc0':
        num = WE2.sum(axis=1)
    elif cov=='hc1':
        num = WE2.sum(axis=1)*(Jc/(Jc-1).where(Jc>=2))
    elif cov=='hc2':
        num = (WE2/(1-h)).sum(axis=1)
    elif cov=='hc3':
        num = (WE2/(1-h)**2).sum(axis=1)

    return (num/den**2).where(Jc>=2)

** Full covariance of $w$
=w_var= gives only the marginal variances (the diagonal of $V_w$).  For
propagating $w$-uncertainty into a downstream generated-regressor problem
the cross-cell covariance is needed too.  Because the $w$-stage is linear
given \beta --- the joint design is $y_0 = a_{tm} + \beta_j w_{itm} + e$
with an $(t,m)$-intercept block $A$ and a \beta-loaded block $B$ --- the
covariance factors and never needs to be formed as the dense
$N_w\times N_w$ matrix.

Writing $d_c=\sum_{j\in O_c}\beta_j^2$, $s_c=\sum_{j\in O_c}\beta_j$, the
$B$-block normal matrix $B'B=\diag(d_c)$ is diagonal and $A'A=\diag(n_{tm})$
counts observations per market-period.  By the partitioned inverse (FWL),
$\Var(\hat w)=\sigma^2(B'M_AB)^{-1}$ with $M_A=I-A(A'A)^{-1}A'$, and Woodbury
gives the low-rank factoring
\[
   V_w = \sigma^2\bigl[\diag(1/d_c) + U\,V_{22}\,U'\bigr],\quad
   U_{c}=s_c/d_c,\quad
   V_{22,tm}=\frac{1}{\,n_{tm}-\sum_{c\in tm}s_c^2/d_c\,},
\]
where the $V_{22}$ block is /diagonal/ in $(t,m)$ (no cross-market
coupling), so cells are correlated only /within/ a market-period, rank-1
each.  The off-diagonal term is positive: fitting the $(t,m)$ intercept out
inflates $\Var(\hat w)$ above the naive $\sigma^2/d_c$.  This is the
homoskedastic covariance with \beta treated as known; for robust marginals
use =w_var= with a =cov= option.
#+name: code:wcov
#+begin_src python
class WCov:
    """Factored sampling covariance of the welfare-weight estimates w_hat.

    Conditional on beta the w-stage is linear, so V_w is a diagonal plus a
    within-(t,m) rank-1 term and is stored by its factors (the dense
    N_w x N_w matrix is never formed):

        V_w = diag(D) + sigma2 * U V22 U',  cells coupled only within (t,m).

    Attributes
    ----------
    sigma2 : float        pooled error variance
    D : Series (i,t,m)     diagonal part, sigma2 / sum_{O_c} beta_j^2
    U : Series (i,t,m)     loadings s_c/d_c, s_c = sum_{O_c} beta_j
    V22 : Series (t,m)     1 / (n_tm - sum_{c in tm} s_c^2/d_c)
    """
    __slots__ = ['sigma2','D','U','V22']

    def __init__(self,sigma2,D,U,V22):
        self.sigma2,self.D,self.U,self.V22 = sigma2,D,U,V22

    def _v22_by_cell(self):
        return self.V22.reindex(self.U.index.droplevel('i')).values

    def var(self):
        """Marginal variances Var(w_c), the diagonal of V_w."""
        return self.D + self.sigma2*self.U**2*self._v22_by_cell()

    def matvec(self,x):
        """Return V_w @ x without forming V_w (x a Series indexed by cell)."""
        x = x.reindex(self.D.index)
        g = (self.U*x).groupby(['t','m']).sum()*self.V22     # v22_tm * sum_c U_c x_c
        gc = g.reindex(self.U.index.droplevel('i')).values
        return self.D*x + self.sigma2*self.U*gc

    def todense(self,tm):
        """Dense covariance block for one market-period tm=(t,m) (small)."""
        idx = self.U.xs(tm,level=['t','m'],drop_level=False).index
        U = self.U.loc[idx].values; D = self.D.loc[idx].values
        return pd.DataFrame(np.diag(D) + self.sigma2*self.V22[tm]*np.outer(U,U),
                            index=idx,columns=idx)


def w_cov(e,beta):
    """Factored covariance of w_hat (homoskedastic, beta treated as known).

    Parameters
    ----------
    e : Series or DataFrame
        Regression residuals (e.g. ``self.e3``); Series indexed (i,t,m,j) or
        a wide (i,t,m) x j DataFrame.
    beta : Series
        Frisch elasticities, indexed by good ``j``.

    Returns
    -------
    WCov
        Factored covariance with ``.var()``, ``.matvec(x)`` and
        ``.todense((t,m))`` helpers.  The diagonal's leading term matches
        ``w_var(e, beta, cov='homoskedastic')``; the extra positive term is
        the cost of estimating the (t,m) intercept.
    """
    E = e.unstack('j') if e.ndim==1 else e
    beta = pd.Series(beta).reindex(E.columns)
    obs = E.notna()
    d = obs.multiply(beta**2,axis=1).sum(axis=1)         # d_c = sum_{O_c} beta_j^2
    s = obs.multiply(beta,axis=1).sum(axis=1)            # s_c = sum_{O_c} beta_j
    Jc = obs.sum(axis=1)                                 # |O_c|
    keep = Jc>=2

    sigma2 = (E**2).sum().sum()/(Jc-1).clip(lower=0).sum()  # matches w_var homoskedastic

    D = (sigma2/d).where(keep)
    U = (s/d).where(keep)

    n_tm = Jc.groupby(['t','m']).sum()                   # total obs per market-period
    corr = (s**2/d).where(d>0).groupby(['t','m']).sum()  # sum_c s_c^2/d_c (saturated cells net to 0)
    V22 = 1/(n_tm - corr)

    return WCov(sigma2,D,U,V22)

def estimate_pi(y,b,w,Ar,gamma_d,verbose=False):

    try:
        y0 = y.stack()
    except AttributeError:
        y0 = y.copy()

    wb = outer(w,b).stack()

    e = y0 - Ar - wb - gamma_d

    e = e.dropna()

    pi_g = e.groupby(['t','m'])

    pi = pi_g.mean()
    pi.name = 'pi'

    pi_se = pi_g.std()/np.sqrt(pi_g.count())

    assert np.all(pi_se>0), "Non-positive estimates of pi_se?!"

    e4 = e - pi
    e4 = e4.reorder_levels(['i','t','m','j']).sort_index()

    return pi, pi_se, e4

** Rectifying the global scale of $w$ (prototype)
=estimate_pi= forces the coefficient on $\beta w$ to one --- it is the
$\delta=1$ restriction of the level/scale regression
\[
   y_{itmj} - (\gamma d)_{itmj} = \pi_{tm} + A(r)_{tmj} + \delta\,(\beta_j w_{itm}) + e_{itmj}.
\]
Freeing $\delta$ pins the /global scale/ of $w$: the SVD scale that $\beta$
carries (from the doubly-residualized covariance) need not be the scale at
which $\beta w$ best reproduces the level equation (built from the
singly-residualized data), and $\delta$ is the correction.  We rescale
$\tilde w=\delta w$ (so $\beta$ keeps its SVD scale).

This iterates, recomputing $A(r)$ each pass (it was built under $\delta=1$):
given $w$, partial the $(t,m)$ level out of both $y-\gamma d-A(r)$ and
$\beta w$ and read off $\delta$ (OLS slope, $A(r)$ held as offset); rescale
$w\leftarrow\delta w$; recompute $A(r)$ as the $(t,m,j)$ mean of
$y-\gamma d-\beta w$, centered over $j$.  At a fixed point $\delta\to1$ and
the cumulative $\Delta=\prod_k\delta_k$ settles.  A sign flip ($\delta\le0$)
raises (the $\beta$-direction is anti-correlated with the $w$-variation ---
something is badly wrong); non-convergence or a $\Delta$ far from 1 warns.

This is a prototype: it is /not/ wired into =get_pi= yet.  Inference on
$\delta$ (a generated-regressor slope; its s.e. follows from =w_cov=) is
deferred --- see notes.

def rectify_w(y,gamma_d,beta,w,Ar,max_iter=20,tol=1e-4,accelerate=True,
              warn_band=None,verbose=False):
    """Estimate a global scale delta on beta*w, iterating with A(r) recomputed.

    Model: y - gamma_d = pi_tm + A(r)_tmj + delta*(beta_j w_itm) + e.

    Convergence is geometric and can be slow (~10-15 passes under heavy
    missingness).  Since the net effect is always w = Delta * w_in and the
    cumulative-Delta sequence converges geometrically, accelerate=True
    (default) applies Aitken delta^2 to extrapolate the limit and jump there
    in a few passes.  Sign flip (delta<=0) raises; non-convergence warns.
    warn_band, if given, warns when the cumulative Delta leaves the band --
    but the economic+statistical gate lives in Regression.rectify (which has
    the delta s.e.).

    Returns
    -------
    (w, Ar, pi, Delta, history, e3)
        Rectified w (= Delta * w_in), recomputed A(r), pi and (t,m,j)-demeaned
        residual e3, the cumulative scale Delta = prod_k delta_k, and the
        per-iteration delta history.
    """
    L = (y - gamma_d).dropna()
    beta = pd.Series(beta).squeeze() if hasattr(beta,'squeeze') else pd.Series(beta)
    w0 = w.copy()                       # net effect is always w = Delta * w0
    Ar = Ar.copy()
    Delta = 1.0
    history = []
    Dseq = []
    aitken_prev = None
    converged = False

    def offset(Ar):
        return pd.Series(Ar.reindex(L.index.droplevel('i')).values,index=L.index)

    def recompute_Ar(w):
        # A(r): (t,m,j) mean of (y - gamma_d - beta*w), centered over j.
        bw = outer(w,beta).stack().reindex(L.index)
        A = (L - bw).groupby(['t','m','j']).mean()
        return A - A.groupby(['t','m']).transform('mean')

    for it in range(max_iter):
        delta = _delta_components(L,beta,w,Ar)[0]   # FWL slope on beta*w (A(r) offset)

        if not np.isfinite(delta):
            raise ValueError(f"rectify: non-finite delta at iter {it}.")
        if delta <= 0:
            raise ValueError(
                f"rectify: delta={delta:.3g} <= 0 at iter {it}; the beta-direction "
                "is anti-correlated with the w-variation -- something is "
                "dramatically wrong upstream (sign of beta or w?).")

        history.append(delta)
        Delta *= delta
        Dseq.append(Delta)
        w = delta*w
        Ar = recompute_Ar(w)

        if verbose:
            print(f"  rectify iter {it}: delta={delta:.6f}  Delta={Delta:.6f}")
        if abs(delta-1) < tol:
            converged = True
            break

        # Aitken delta^2 on the (geometrically convergent) cumulative-Delta
        # sequence: extrapolate the limit and jump w = Delta_inf * w0.
        if accelerate and len(Dseq) >= 3:
            d0,d1,d2 = Dseq[-3:]
            denom = d2 - 2*d1 + d0
            if abs(denom) > 1e-12:
                aitken = d0 - (d1-d0)**2/denom
                if (1e-3 < aitken < 1e3 and aitken_prev is not None
                        and abs(aitken-aitken_prev) < tol):
                    Delta = aitken
                    w = Delta*w0
                    Ar = recompute_Ar(w)
                    converged = True
                    if verbose:
                        print(f"  rectify: Aitken-accelerated to Delta={Delta:.6f} "
                              f"at iter {it}")
                    break
                aitken_prev = aitken

    if not converged:
        warnings.warn(
            f"rectify did not converge in {max_iter} iterations "
            f"(last delta={history[-1]:.4g}, Delta={Delta:.4g}).")

    if warn_band is not None and not (warn_band[0] <= Delta <= warn_band[1]):
        warnings.warn(
            f"rectify: cumulative Delta={Delta:.4g} outside {warn_band}; "
            "large global w-rescaling -- check specification.")

    bw = outer(w,beta).stack().reindex(L.index)
    resid = L - offset(Ar) - bw
    pi = resid.groupby(['t','m']).mean()
    pi.name = 'pi'

    pi_b = pd.Series(pi.reindex(L.index.droplevel(['i','j'])).values,index=L.index)
    e = resid - pi_b
    e3 = e - e.groupby(['t','m','j']).transform('mean')

    return w,Ar,pi,Delta,history,e3

*** Inference on $\delta$ (generated-regressor s.e. via =w_cov=)
The one-pass slope $\hat\delta=(z'u)/(z'z)$ --- with $u=M_{tm}(y-\gamma d-A(r))$
and $z=M_{tm}(\beta w)$ --- is a /generated-regressor/ statistic: the
regressor $z$ is built from the estimated $\hat w$.  Two channels feed
$\Var(\hat\delta)$:
1. the level-regression noise (the usual OLS term, "naive"); and
2. the sampling error in $\hat w$, whose covariance $V_w$ we already have
   from =w_cov=.

Conditioning on $A(r)$, $\hat\delta$ is a smooth function of the cell vector
$\hat w$, so the delta method gives
\[
   \Var(\hat\delta) \approx \underbrace{\frac{\sigma_e^2}{z'z}}_{\text{naive}}
      \;+\; g'\,V_w\,g,\qquad
   g_a=\frac{\partial\hat\delta}{\partial \hat w_a},
\]
with (writing $s_c=\sum_{j\in O_c}\beta_j$, $n_{tm}$ the obs count, and
$\phi_{cj}=(u_{cj}-2\hat\delta z_{cj})/z'z$)
\[
   g_a=\sum_{j\in O_a}\beta_j\phi_{aj}
       \;-\;\frac{s_a}{n_{tm(a)}}\sum_{c\in tm(a)}\sum_j\phi_{cj}.
\]
The quadratic form $g'V_wg$ uses the =w_cov= factors directly (no dense
matrix).  The statistic $z=(\hat\delta-1)/\se(\hat\delta)$ tests the null
that $\beta w$ at its SVD scale needs no rescaling --- i.e. a specification
check that the two residualizations agree.  (This is the one-pass test;
inference on the /iterated/ cumulative $\Delta$ is a further step.)

from collections import namedtuple

WDelta = namedtuple('WDelta',['delta','se','se_naive','se_gen','z','grad'])
RectifyResult = namedtuple('RectifyResult',['Delta','applied','wd','history'])

def _delta_components(L,beta,w,Ar):
    """One-pass delta and the demeaned pieces. L = (y - gamma_d).dropna()."""
    def offset(A):
        return pd.Series(A.reindex(L.index.droplevel('i')).values,index=L.index)
    bw  = outer(w,beta).stack().reindex(L.index)
    lhs = L - offset(Ar)
    u = lhs - lhs.groupby(['t','m']).transform('mean')   # M_tm(y - gamma_d - A(r))
    z = bw  - bw.groupby(['t','m']).transform('mean')    # M_tm(beta*w)
    den = (z**2).sum()
    delta = (u*z).sum()/den
    return delta,u,z,den

def w_delta(y,gamma_d,beta,w,Ar,V=None):
    """One-pass global scale delta on beta*w, with generated-regressor s.e.

    Parameters
    ----------
    V : WCov or None
        If given (e.g. from ``w_cov``), add the propagated uncertainty in the
        estimated w to the naive level-regression variance.  Conditions on
        A(r); see notes.  Cells with <2 observed goods have undefined V_w and
        do not contribute to the generated-regressor term, so se_gen is mildly
        understated (the delta=1 test slightly anti-conservative) under heavy
        missingness.

    Returns
    -------
    WDelta(delta, se, se_naive, se_gen, z, grad)
        z = (delta-1)/se tests the null that the SVD scale of w is right.
    """
    L = (y - gamma_d).dropna()
    beta = pd.Series(beta).squeeze() if hasattr(beta,'squeeze') else pd.Series(beta)
    delta,u,z,den = _delta_components(L,beta,w,Ar)

    e = u - delta*z                                   # level-regression residual
    sigma2 = (e**2).sum()/(e.shape[0]-1)
    var_naive = sigma2/den

    var_gen = 0.0
    grad = None
    if V is not None:
        bj = pd.Series(beta.reindex(u.index.get_level_values('j')).values,index=u.index)
        phi = (u - 2*delta*z)/den
        G      = (phi*bj).groupby(['i','t','m']).sum()         # sum_j beta_j phi_cj
        gcell  = phi.groupby(['i','t','m']).sum()              # sum_j phi_cj
        s      = bj.groupby(['i','t','m']).sum()               # s_c = sum_{O_c} beta_j
        ntm    = u.groupby(['t','m']).size()                   # n_tm
        tm_sum = gcell.groupby(['t','m']).sum()                # sum_{c in tm} sum_j phi_cj
        tm = G.index.droplevel('i')
        grad = G - s*(tm_sum.reindex(tm).values)/(ntm.reindex(tm).values)
        # g' V_w g via factors (cells with undefined V_w contribute 0)
        D  = V.D.reindex(grad.index).fillna(0.0)
        Uc = V.U.reindex(grad.index).fillna(0.0)
        gtm = (Uc*grad).groupby(['t','m']).sum()
        var_gen = (D*grad**2).sum() + V.sigma2*(V.V22*gtm**2).sum()

    se = np.sqrt(var_naive + var_gen)
    zstat = (delta-1)/se if se>0 else np.nan
    return WDelta(delta,se,np.sqrt(var_naive),np.sqrt(var_gen),zstat,grad)

    def get_beta(self,verbose=None,compute_se=None,bootstrap_tol=1e-2,heteroskedastic=False):
        """
        Return beta.

        If compute_se=True, compute standard errors and the variance matrix
        as a side-effect (via a resampling bootstrap in estimate_beta).

        WARNING (nested-bootstrap footgun): the compute_se=True path runs an
        inner >=30-iteration bootstrap.  When resampling the whole pipeline
        in an outer loop, call with compute_se=False (the point-estimate
        path) so the beta bootstrap does not nest ~30x inside each outer
        draw.  compute_se defaults to self.attrs['compute_se'].
        """

        if 'Mpd' in self.attrs and self.attrs['Mpd']:
            residualize = self.get_Mpdy
        else:
            residualize = self.get_MpMdy

        if compute_se is None:
            compute_se = self.attrs['compute_se']

        if not compute_se:
            if self.beta is not None:
                return self.beta

        if compute_se:
            if self.beta_se is not None:
                return self.beta # Always return beta!

        # If we get here, we want to compute beta_se, and haven't yet.

        if verbose is None:
            verbose = self.attrs['verbose']

        b, seb, V = estimate_beta(residualize(),verbose=verbose,return_se=compute_se,bootstrap_tol=bootstrap_tol,heteroskedastic=heteroskedastic)
        b = b.squeeze()
        if seb is not None:

            self.beta_V = V

        self.beta = b

        return self.beta

    def w_se(self,cov=None):
        """Compute standard errors of estimated ws.

        NOTE: leaving ``cov`` unset selects the legacy ``'hc0'`` estimator and
        raises a DeprecationWarning.  HC0 is biased *low* at finite J because
        the w-residuals come from a projection (the beta-direction is fitted
        out): coverage ratios run ~0.78 with complete data and worse under
        missing goods (~0.6 at 30% missing, ~0.4 at 50%).  It is retained as
        the default only for continuity with existing downstream code.  Pass
        ``cov='hc2'`` for the leverage-corrected estimator, which is unbiased
        and remains heteroskedasticity-robust.

        - cov :: Per-good error-variance estimator passed to ``w_var``; one
          of 'hc0' (legacy default), 'hc1', 'hc2' (recommended), 'hc3',
          'homoskedastic', or 'pooled'.
        """
        if cov is None:
            warnings.warn(
                "w_se default cov='hc0' is biased low at finite J and is slated "
                "to change to 'hc2'.  Pass cov explicitly to silence this "
                "warning: 'hc2' (recommended) or 'hc0' (legacy behavior).",
                DeprecationWarning, stacklevel=2)
            cov = 'hc0'

        beta = self.get_beta()

        if self.e3 is None: self.get_w()

        v = w_var(self.e3,beta,cov=cov)

        return np.sqrt(v)

    def rectify(self,inplace=False,econ_tol=0.1,z_tol=3.0,
                max_iter=20,tol=1e-4,verbose=False):
        """PROTOTYPE: pin the global scale of w, gated on BOTH significance.

        Iterates a global slope delta on beta*w (recomputing A(r) and pi each
        pass; see rectify_w) for the point estimate Delta, and computes the
        one-pass delta with its generated-regressor s.e. (see delta_test) for
        the statistical test.  The rescaling is treated as warranted only when
        it is both

          - economically significant : |Delta - 1| > econ_tol, and
          - statistically significant: |z| = |delta-1|/se > z_tol,

        guarding against both negligible-but-real and large-but-noisy
        corrections.  With inplace=True the rescaling w_tilde = Delta*w (and
        recomputed A(r), pi, e3) is written back ONLY if the gate passes;
        otherwise w is left on its SVD scale.  A warning fires when the gate
        passes (a notable correction is being applied).

        Returns RectifyResult(Delta, applied, wd, history) where wd is the
        WDelta from the one-pass test.
        """
        beta = self.get_beta(verbose=verbose)
        gamma_d = self.get_gamma_d(verbose=verbose)
        w = self.get_w(verbose=verbose)
        Ar = self.get_Ar(verbose=verbose)

        # statistical test (one-pass delta with generated-regressor s.e.)
        wd = w_delta(self.y,gamma_d,beta,w,Ar,V=w_cov(self.e3,beta))

        # point estimate (iterated cumulative Delta)
        w2,Ar2,pi2,Delta,history,e3 = rectify_w(self.y,gamma_d,beta,w,Ar,
                                                max_iter=max_iter,tol=tol,verbose=verbose)

        economic    = abs(Delta-1) > econ_tol
        significant = np.isfinite(wd.z) and abs(wd.z) > z_tol
        applied = bool(economic and significant)

        if applied:
            warnings.warn(
                f"rectify: applying w-rescaling Delta={Delta:.4g} "
                f"(|Delta-1|={abs(Delta-1):.3g} > {econ_tol}, "
                f"|z|={abs(wd.z):.3g} > {z_tol}).")
            if inplace:
                self.w = w2
                self.Ar = Ar2
                self.pi = pi2
                self.e3 = e3          # so w_se/w_cov reflect rectified w
                self.yhat = None      # predictions stale
                self.e4 = None
        elif verbose:
            why = ("economically negligible" if not economic else
                   "not statistically significant")
            print(f"  rectify: no rescaling ({why}); "
                  f"Delta={Delta:.4g}, z={wd.z:.3g}")

        return RectifyResult(Delta,applied,wd,history)

    def delta_test(self):
        """PROTOTYPE: one-pass global scale delta with generated-regressor s.e.

        Propagates the uncertainty in the estimated w (via w_cov) into the
        slope on beta*w, conditioning on A(r).  Returns a WDelta; the z-stat

        gamma_d = self.get_gamma_d()

    def predicted_expenditures(self,resample=False,clusterby=['t','m','j'],fill_missing=True,verbose=False):
        """Compute predicted /levels/ of expenditures.

        This is different from exp(yhat), since we have to account for the expected value of exp(e), where e = y - yhat.

        One standard (and the default) way to calculate these is by assuming that the distribution of e is normal.  An alternative is to resample residuals.

        Regardless of whether resampling is chosen, means (and variances) are selected at the level of the list clusterby.
        """
        if self.yhat is None:
            self.yhat = self.get_predicted_log_expenditures(fill_missing=True,verbose=verbose)

        e = self.y - self.yhat
        eg = e.dropna().groupby(clusterby)

        if not resample:
            # Use iqr instead of variance for some robustness to outliers
            # Relation for normal dist: iqr/1.3489795 = sigma
            evar = (eg.transform(iqr)/1.3489795)**2
        else:
            if resample < 1: # Assume this is a tolerance
                tol = resample
            last = -1
            evar = e.dropna().groupby(clusterby).transform(np.var)
            evar = evar.sort_index(level=clusterby)
            i = 0
            diff = 1
            while diff>tol:
                last = evar
                esample = eg.sample(frac=1,replace=True)
                drawvar = (esample.groupby(clusterby).transform(iqr)/1.3489795)**2
                evar = i/(i+1)*evar + drawvar.values/(i+1)
                i += 1
                diff = np.abs(evar-last).max()
                if verbose: print(f'Draw {i}, diff={diff}')

        if fill_missing:
            evar = evar.groupby(clusterby).mean()
            xhat = np.exp(self.yhat.add(evar/2))
            xhat = xhat.reorder_levels(self.yhat.index.names).sort_index()
        else:
            xhat = np.exp(self.yhat + evar/2)
            xhat = xhat.reorder_levels(self.yhat.index.names).sort_index()

        return xhat

Here we want estimates of \lambda, or the antilog of $-w$.  We have a CLT for $w$ [cite:@bai03] and an expression for the asymptotic variance of $w$, say $\hat\sigma^2$.  Accordingly, we should have $\lambda = e^{-w}\approx \E e^{-\hat w + \hat\sigma^2/2}$.

        """Expected value of minus the antilog of $w$.

        lhat = np.exp(-self.w + v/2)

        return lhat

It's possible to estimate =w= without first estimating \beta, etc. or using factor analysis.  We simply compute =Mpdy= (residuals from the SUR regression), then compute the mean across $j$ for every $(i,t,m)$.  The defect in this simple plan is that it relies on all households reporting positive expenditures for the same number of goods.  Still, this may not matter than much in practice.  It should be more efficient to weight by the reciprocal of the standard deviation of errors for each good, using a GLS sort of logic; this is made an option.

        else:
            sigmaj = 1
        w = (e/sigmaj).groupby(['i','t','m']).mean()

def _demand_parameters(self,p=None,d=None):
    """Return tuple of p and dictionary of (alpha,beta,phi) from regression instance.

    Suitable for passing to =cfe.demand= functions.
    """

    beta = self.beta
    n = len(beta)

    if d is None:
        gd = self.get_gamma_d().groupby('j').mean()
    else:
        gd = d@self.gamma.T

    alpha = np.exp(gd)

    if p is None or len(p)==0:
        prices = np.exp((self.pi + self.Ar).groupby('j').mean())
    else:
        prices = p

    assert len(prices), f"What happened to prices? p={prices}."

    phi = 0 # phi not (yet?) an attribute of Regression.

    # Cast to numeric arrays
    try:
        prices = prices.values
    except AttributeError:
        pass

    try:
        alpha = alpha.values
    except AttributeError:
        pass
    try:
        beta = beta.values
    except AttributeError:
        pass

    try:
        phi = phi.values
    except AttributeError:
        pass

    return prices,{'alpha':alpha,'beta':beta,'phi':phi}

def _cells(rows):

        rows,

def test_hc0_matches_sandwich_scale():
    # The corrected HC0 must equal the OLS-slope sandwich for w_hat_c =
    # (beta'X_c)/(beta'beta): sum_j beta_j^2 e_cj^2 / (sum_j beta_j^2)^2.
    beta = pd.Series([1.,2.,3.], index=pd.Index([0,1,2],name='j'))
    e = _cells([[0.1,-0.2,0.05],[0.0,0.1,-0.1]])
    v = w_var(e, beta, cov='hc0')

    E = e.unstack('j').values; b = beta.values; b2 = (b**2).sum()
    expect = (b**2*E**2).sum(1)/b2**2
    assert np.allclose(v.values, expect)
    # And it is NOT the old (global-betabeta, mean-over-j) formula unless
    # mean(beta_j^2)==1; here mean(beta^2)=14/3, so they must differ.
    old = (E*b)**2
    old = old.mean(1)/b2
    assert not np.allclose(v.values, old)

def test_coverage_hc2_unbiased_hc0_low():

def _brute_force_w_cov(eS, beta, tmlist):
    # Build the joint design [A | B] explicitly and form sigma^2 (B'M_A B)^{-1}
    # for a fully-observed dataset, as a check on the factored w_cov.
    E = eS.unstack('j'); cells = E.index; J = E.shape[1]
    b = beta.reindex(E.columns).values
    cell_tm = [(t,m) for (i,t,m) in cells]
    nobs = len(cells)*J
    A = np.zeros((nobs, len(tmlist))); B = np.zeros((nobs, len(cells)))
    y = np.zeros(nobs); r = 0
    for ci,(i,t,m) in enumerate(cells):
        for jj in range(J):
            A[r, tmlist.index((t,m))] = 1.0
            B[r, ci] = b[jj]
            r += 1
    MA = np.eye(nobs) - A@np.linalg.inv(A.T@A)@A.T
    BMB = B.T@MA@B
    from cfe.regression import w_cov as _wc
    sigma2 = _wc(eS, beta).sigma2
    return sigma2*np.linalg.inv(BMB), cells

    numerical = (dp - dm)/(2*h)
    analytic = float((wd.grad*v).sum())
    assert abs(numerical - analytic) < 1e-4*(abs(analytic)+1), (numerical, analytic)


def test_w_delta_gen_variance_equals_dense_quadratic_form():
    # var_gen (factor form) must equal grad' V_w grad (dense) on clean data.
    rng = np.random.default_rng(9)
    y,gd,beta,w,Ar = _exact_w_model(N=150, J=5)
    y = y + pd.Series(rng.standard_normal(len(y))*0.3, index=y.index)
    gd = y*0
    V = w_cov((y-gd), beta)
    assert not V.D.isna().any()                # complete data -> no undefined cells
    wd = w_delta(y, gd, beta, w, Ar, V=V)
    Vd = V.todense((0,0)); g = wd.grad.reindex(Vd.index).values
    assert np.isclose(wd.se_gen, np.sqrt(g @ Vd.values @ g))


def test_w_delta_se_components():
    y,gd,beta,w,Ar = _messy_w_model()
    V = w_cov((y-gd), beta)
    wd = w_delta(y, gd, beta, w, Ar, V=V)
    # combined se exceeds the naive (generated-regressor adds variance)
    assert wd.se >= wd.se_naive - 1e-12
    assert wd.se_gen > 0
    assert np.isclose(wd.se, np.sqrt(wd.se_naive**2 + wd.se_gen**2))
    # complete, exact data -> delta==1 and z ~ 0
    y2,gd2,b2,w2,Ar2 = _exact_w_model()
    wd2 = w_delta(y2, gd2, b2, w2, Ar2, V=w_cov((y2-gd2), b2))
    assert abs(wd2.delta - 1) < 1e-6
